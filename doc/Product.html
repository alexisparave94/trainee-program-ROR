def save_change_log
  fields_and_values = changes.except('updated_at')
  fields_and_values.each do |key, value|
    ChangeLog.create(
      user: User.give_user,
      description: 'Update',
      product: name, field: key, previous_content: value[0], new_content: value[1]
    )
  end
end
